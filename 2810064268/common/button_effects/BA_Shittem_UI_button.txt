		always = yes

		always = yes
	}
	effect = {
		custom_tooltip = "BA_Shittem_gray_button_tip"
		from = {
			hidden_effect = {
				if = {
					limit = {
						check_variable = {
							which = BA_cheat
							value = 5
						}
					}
					country_event = {
						id = BA_Shittem.1998
					}
					set_variable = {

					else = {
						change_variable = {
							which = BA_cheat
							value = 1
						}
					}
				}
			}
		}
	}
