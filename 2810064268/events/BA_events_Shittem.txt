country_event = {
	#调试打招呼用，已限定仅BA政体可以触发，之后会改为触发器
	id = BA_Shittem.0001
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_ai = no
		has_authority = auth_kivotos
	}
	immediate = {
		set_global_flag = Kivotos_detected
		set_country_flag = Kivotos_detected		#一系列突发事件触发用的flag，也是BA堕落帝国生成事件用的flag
		set_country_flag = situation_machine_uprising_started		#防机械叛乱
		set_country_flag = synth_detector		#防机械叛乱
		#创建Arona通讯界面国家
		if = {
			limit = {
				NOT = {
					any_country = {
						has_country_flag = BA_Shittem_country
					}
				}
			}
			create_country = {
				name = "Shittem.name"
				type = Shittem_communication
				flag = {
					icon = {
						category = "special"
						file = "flag_Shittem.dds"
					}
					background = {
						category = "backgrounds"
						file = "00_solid.dds"
					}
					colors = {
						"blue"
						"blue"
						"null"
						"null"
					}
				}
			}
			last_created_country = {
				set_country_flag = BA_Shittem_country
				save_global_event_target_as = BA_Shittem_country
			}
		}
		#设置特殊领袖用种族、复活计数器
		create_species = {
			name = "species_name_BA_leader"
			class = Schale
			portrait = "Schale"
			traits = random
			name_list = "Schale"
			effect = {
				save_global_event_target_as = BA_leader_species
			}
		}
		set_variable = {
			which = BA_leader_respawning
			value = 0
		}
		set_variable = {
			which = BA_leader_respawn_cost
			value = 0
		}
		#设置姬器人种族
		if = {
			limit = {
				NOT = {
					any_galaxy_species = {
						has_species_flag = BA_Robot_flag
					}
				}
			}
			create_species = {
				name = "BA_Robot"
				class = MACHINE
				portrait = "Aris"
				namelist = "ArisName"
				traits = {
					trait = trait_test
					trait = trait_mechanical
					trait = trait_robot_emotion_emulators
					trait = trait_robot_logic_engines
					trait = trait_robot_luxurious
					trait = trait_robot_high_bandwidth
				}
				effect = {
					set_species_flag = BA_Robot_flag
					save_global_event_target_as = BA_Robot_species
				}
			}
		}
		#委托参数
		set_variable = {
			which = BA_commmission_count
			value = 0
		}
		set_variable = {
			which = BA_Repeatable_commmission_01_counter
			value = 1000
		}
		set_variable = {
			which = BA_Repeatable_commmission_02_counter
			value = 1500
		}
		set_variable = {
			which = BA_Repeatable_commmission_03_counter
			value = 50
		}
		#社团限制
		set_variable = {
			which = BA_Club_count
			value = 0
		}
		set_variable = {
			which = BA_Club_limit
			value = 1
		}
		set_variable = {
			which = BA_Club_check
			value = BA_Club_limit
		}
		#目标准备
		capital_scope = {
			#设首都旗帜
			set_planet_flag = BA_Kivotos_capital
			set_planet_flag = colony_event			#屏蔽地底人一类的事件
			#为摇篮计划准备目标（以后这种事尽量放到项目事件里 T_T）
			solar_system = {
				set_star_flag = BA_Project_Cradle_system
				if = {
					limit = {
						NOT = {
							any_system_planet = {
								has_planet_flag = BA_Project_Cradle_target
							}
						}
					}
					random_system_planet = {
						limit = {
							is_star = no
							NOR = {
								is_planet_class = pc_BA_campus_planet
								is_planet_class = pc_BA_ruined_planet
							}
						}
						set_planet_flag = BA_Project_Cradle_target
						set_planet_flag = colony_event
						change_pc = pc_shielded
					}
				}
				if = {
					limit = {
						NOT = {
							any_system_planet = {
								has_planet_flag = BA_Gematria_station
							}
						}
					}
					random_system_planet = {
						limit = {
							is_star = no
							NOR = {
								is_planet_class = pc_BA_campus_planet
								is_planet_class = pc_BA_ruined_planet
								has_planet_flag = BA_Project_Cradle_target
							}
						}
						set_planet_flag = BA_Gematria_station
						#save_global_event_target_as = Project_Cradle_02_target
					}
				}
			}
		}
		#面纱相关
		if = {
			limit = {
				has_valid_civic = civic_Omnia_Vanitas
			}
			set_country_flag = BA_Omnia_Vanitas_under_guise
		}
		#非特有起源的arona
		country_event = {
			id = BA_Shittem.0015
			days = 360
			random = 360
		}
		#科技
		if = {
			limit = {
				NOT = {
					has_origin = origin_Kivotos_Kyrie
				}
			}
			give_technology = {
				tech = tech_BA_legacy_of_millennium
				message = no
			}
		}
	}
}

	is_triggered_only = yes

	diplomatic = yes
	custom_gui = "BA_window"
	picture_event_data = {
		portrait = Arona0022
		room = "BA_Shittem_room"
	}
	#返回
	option = {
		name = BA_Shittem.1002.exit
	}
	#重新复活领袖
	option = {
		name = Respawn_leader
		response_text = Respawn_leader_re
		custom_tooltip = Respawn_leader_tip
		trigger = {
			has_technology = tech_BA_pyroxenes_crafting
			check_variable = {
				which = BA_leader_respawning
				value > 0
			}
		}
		allow = {
			resource_stockpile_compare = {
				resource = sr_pyroxenes
				value >= 500
				multiplier = BA_leader_respawning
			}
		}
		hidden_effect = {
			country_event = {
				id = BA_Shittem.1965
				days = 15
			}
			set_variable = {
				which = BA_leader_respawning
				value = 0
			}
		}
	}
	#摇篮计划
	option = {
		name = BA_Shittem.1002.ProjectCradle
		trigger = {
			#有屏障星球
			any_planet_within_border = {
				has_planet_flag = BA_Project_Cradle_target
				is_planet_class = pc_shielded
			}
			#完成后隐藏
			NOR = {
				has_country_flag = BA_Project_Cradle_Engaging
				has_country_flag = BA_Project_Cradle_finished
			}
		}
		allow = {
			fail_text = {
				text = BA_Shittem.1002.ProjectEngaging
				#没有进行中的其他项目
				# NOT = {
				# 	has_global_flag = BA_PROJECT_CRADLE
				# }
			}
		}
		custom_tooltip = BA_Project_Cradle_instruction
		hidden_effect = {
			country_event = {
				id = BA_Project.0010
			}
			#开始项目！
			set_country_flag = BA_Project_Cradle_Engaging
		}
	}
	#请求Arona帮助停止废墟产生
	option = {
		name = Kyrie_ruins
		is_dialog_only = yes
		response_text = Kyrie_ruins.response
		trigger = {
			any_owned_planet = {
				has_planet_flag = BA_Kyrie_target_@root
				has_planet_flag = BA_ruins_happening
			}
		}
		allow = {
			custom_tooltip = {
				text = Kyrie_ruins.tip
				resource_stockpile_compare = {
					resource = sr_pyroxenes
					value >= 50
				}
				NOT = {
					has_country_flag = BA_Kyrie_Arona_will_help_you
				}
			}
		}
		add_resource = {
			sr_pyroxenes = -50
		}
		set_timed_country_flag = {
			flag = BA_Kyrie_Arona_will_help_you
			days = 1800
		}
	}
}
